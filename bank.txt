pragma solidity ^0.8.0;

contract bank{
    uint256 balance=0;
    address public accOwner;
    
    constructor(){
        accOwner=msg.sender;
    }

    // Deposite money
    function deposite() public payable{
        require(accOwner==msg.sender,"You are not account owner");
        require(msg.value>0,"Amount must be greater than 0");
        balance+=msg.value;
    
    }

    // withdraw money
    function withdraw(uint256 amt)public payable {
        require(accOwner==msg.sender,"You are not account owner");
        require(amt>0,"Amount must be greater than 0");
        require(amt<=balance,"Insufficient Balance");
        balance-=amt;
        payable(msg.sender).transfer(amt);
    }

    // show balance
    function show_balance()public view returns(uint256){
        require(accOwner==msg.sender,"You are not account owner");
        return balance;
    }
}
